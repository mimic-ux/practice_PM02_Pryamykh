11.
SELECT
    p.name,
    p.article,
    COALESCE(SUM(i.quantity), 0) - COALESCE(SUM(o.quantity), 0) AS balance
FROM Products p
LEFT JOIN Incoming i ON p.id_product = i.id_product
LEFT JOIN Outgoing o ON p.id_product = o.id_product
GROUP BY p.id_product;


12.
SELECT i.date, p.name AS product_name, i.quantity, i.price
FROM Incoming i
JOIN Products p ON i.id_product = p.id_product
JOIN Suppliers s ON i.id_supplier = s.id_supplier
WHERE s.name = 'ООО МеталлТорг';


13.
SELECT
    p.name,
    COALESCE(SUM(i.quantity), 0) - COALESCE(SUM(o.quantity), 0) AS balance
FROM Products p
LEFT JOIN Incoming i ON p.id_product = i.id_product
LEFT JOIN Outgoing o ON p.id_product = o.id_product
GROUP BY p.id_product
ORDER BY balance DESC;

19.

FROM Incoming i
JOIN Suppliers s ON i.id_supplier = s.id_supplier

24.
CREATE VIEW StockBalance AS
SELECT p.id_product, p.name,
       COALESCE(SUM(i.quantity), 0) - COALESCE(SUM(o.quantity), 0) AS balance
FROM Products p
LEFT JOIN Incoming i ON p.id_product = i.id_product
LEFT JOIN Outgoing o ON p.id_product = o.id_product
GROUP BY p.id_product;

25.
SELECT p.name,
       COALESCE(SUM(i.quantity), 0) AS total_incoming_qty,
       COALESCE(SUM(o.quantity), 0) AS total_outgoing_qty,
       COALESCE(SUM(i.quantity), 0) - COALESCE(SUM(o.quantity), 0) AS current_balance,
       CASE WHEN SUM(i.quantity) > 0 THEN SUM(i.quantity * i.price) / SUM(i.quantity)
            ELSE NULL END AS avg_purchase_price
FROM Products p
LEFT JOIN Incoming i ON p.id_product = i.id_product
LEFT JOIN Outgoing o ON p.id_product = o.id_product
GROUP BY p.id_product;
